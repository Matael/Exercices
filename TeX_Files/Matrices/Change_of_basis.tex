\bexo
What is the image $[\tb{M}']$ of 
$[\tb{M}]=\left[
\begin{array}{cc}
2 & -1 \\
-1 & 2
\end{array}
\right]
$
 in the 
$\left[
\begin{array}{cc}
1 & 1 \\
1 & -1
\end{array}
\right]
$ basis

\eexo\solution{
\begin{equation*}
[\tb{M}']=\left[
\begin{array}{cc}
3 & 0 \\
0 & 1
\end{array}
\right]
\end{equation*}
}

\bexo
What is the image $[\tb{M}']$ of 
$[\tb{M}]=\left[
\begin{array}{cc}
1 & 1 \\
1 & -1
\end{array}
\right]
$
 in the 
$\left[
\begin{array}{cc}
\dfrac{\sqrt{2}}{2} & \dfrac{\sqrt{2}}{2} \\
\dfrac{\sqrt{2}}{2} & -\dfrac{\sqrt{2}}{2}
\end{array}
\right]
$ basis

\eexo\solution{
\begin{equation*}
[\tb{M}']=\left[
\begin{array}{cc}
1 & 1 \\
1 & -1
\end{array}
\right]
\end{equation*}
}

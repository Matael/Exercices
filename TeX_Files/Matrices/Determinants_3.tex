\bexo
What is the determinant of 
\begin{equation}
\mat{M}=
\begin{bmatrix}
	1& 2& 0& 0\\
2& 1& 0& 0\\
0& 0& 1& 1\\
0& 0& 1& -1
\end{bmatrix} ?
\end{equation}
\eexo

\solution{
\begin{equation}
	\det{\mat{M}}=\begin{vmatrix}
1 & 2\\ 2 & 1	
\end{vmatrix}\times
\begin{vmatrix}
1 & 1\\ 1 & -1	
\end{vmatrix}=(-3)\times(-2)=6
\end{equation}
}

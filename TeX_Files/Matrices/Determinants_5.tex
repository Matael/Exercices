\bexo
What is the determinant of 
\begin{equation}
\mat{M}=
\begin{bmatrix}
	2& -1& 0\\
-1& 2& -1\\
0& -1& 2
\end{bmatrix} ?
\end{equation}
\eexo

\solution{
\begin{equation}
	\det{\mat{M}}=2
	\begin{vmatrix}
		2 & -1\\-1 & 2
	\end{vmatrix}-(-1)
		\begin{vmatrix}
		-1 & -1\\0 & 2
	\end{vmatrix}=6-2=4
\end{equation}
}

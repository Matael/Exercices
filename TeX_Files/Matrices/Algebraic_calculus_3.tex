\bexo
Compute $\mat{M}^n\tb{X}$ for
\begin{equation}
	\mat{M}=\begin{bmatrix}
		2 & -1\\-1 &2
	\end{bmatrix} \and \tb{X}=\begin{Bmatrix}
		1\\-1
	\end{Bmatrix}
\end{equation}
\eexo 



\solution{
\begin{equation}
\mat{M}\tb{X}=\begin{Bmatrix}
		3\\-3
	\end{Bmatrix}=3\tb{X}.
\end{equation}
\tb{X} is then an eigenvector associated to eigenvalue $\lambda=3$.
\begin{equation}
\mat{M}^2\tb{X}=3^2\tb{X} \so
\mat{M}^n\tb{X}=3^n\tb{X}=\begin{Bmatrix}
		3^n\\-3^n
	\end{Bmatrix}
.
\end{equation}

}

\bexo
What are the eigenvalues of 
\begin{equation*}
\left[
\begin{array}{cc}
1 & 2 \\
0 & 2
\end{array}
\right]
\end{equation*}
For each eigenvalue $\lambda$, propose one eigenvector $\tb{X}$
\eexo\solution{
\begin{equation*}
\lambda=1 \esp \tb{X}=\left\{
\begin{array}{c}
1\\
0
\end{array}
\right\}
\end{equation*}
\begin{equation*}
\lambda=2 \esp \tb{X}=\left\{
\begin{array}{c}
2\\
1
\end{array}
\right\}
\end{equation*}
}

\bexo
What are the eigenvalues of 
\begin{equation}
\begin{bmatrix}
1 & 0 & 0\\
0 & 1 & 0\\
1 & -1 & 2
\end{bmatrix}
\end{equation}
For each eigenvalue $\lambda$, propose one eigenvector $\tb{X}$
\eexo
\solution{
\begin{equation}
\lambda=1 \esp \tb{X}_1=\left\{
\begin{array}{c}
1\\
1\\
0
\end{array}
\right\}
\esp
\tb{X}_2=\left\{
\begin{array}{c}
0\\
1\\
1
\end{array}
\right\}
\end{equation}
\begin{equation}
\lambda=2 \esp \tb{X}=\left\{
\begin{array}{c}
0\\
0\\
1
\end{array}
\right\}
\end{equation}}

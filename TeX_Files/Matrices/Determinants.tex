\bexo
What is the determinant of 
\[
\left[
\begin{array}{cccc}
a_1 & a_2 & \cdots & a_n\\
a_1 & a_1 & \ddots & \vdots\\
\vdots & \ddots & \ddots & a_2\\
a_1 & \cdots & a_1 & a_1
\end{array}
\right]
\]
\eexo\solution{
\begin{equation*}
a_1(a_1-a_2)^{n-1}
\end{equation*}
}

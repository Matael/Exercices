\bexo


\begin{itemize}
	\item What is the Taylor expansion at order $2$ of the $\cos$ function in $x=0$ ?
	\item What is the Taylor expansion at order $2$ of the $\cos$ function in $x=\dfrac{\pi}{4}$ ?
\end{itemize}
 

\eexo
\solution{
\begin{itemize}
	\item $\cos(x)\approx 1-\dfrac{x^2}{2}+\o(x^2) $
	\item $\cos(x)\approx \dfrac{\sqrt{2}}{2}\left[1-\left(x-\dfrac{\pi}{4}\right)
 -\dfrac{\left(x-\dfrac{\pi}{4}\right)^2}{2}\right]+\o\left(\left(x-\dfrac{\pi}{4}\right)^2\right)$
\end{itemize}

}

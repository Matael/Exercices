\bexo
Let $[\tb{M}]$ be a $n\times n $ matrix. Show that the solution of 
\begin{equation}\label{eq:ex linalg 4}
[\tb{M}]\tb{X}=\tb{0}
\end{equation}
is a vector space.
\eexo
\solution{
Let $X_1$ and $X_2$ be two solutions of \refeq{eq:ex linalg 4}:
\begin{equation}
[\tb{M}]\tb{X}_1=\tb{0} \esp 
[\tb{M}]\tb{X}_2=\tb{0}.
\end{equation}
By adding these two equations 
\begin{equation}
	[\tb{M}](\tb{X}_1+\tb{X}_2)=\tb{0}.
\end{equation}
Let $\lambda$ be a real number
\begin{equation}
	[\tb{M}](\lambda\tb{X}_1)=\tb{0}.
\end{equation}
Hence $\lambda \tb{X}_1$ is solution of \refeq{eq:ex linalg 4}. 

}

\bexo
On considère la fonction
\begin{equation}
	f:x\mapsto (x-1)(x+2)(x-3)
\end{equation}
\begin{itemize}
	\item Développez la fonction $f$
	\item Donner l'expression de la dérivée $f'$ de la fonction $f$.
	\item Donner l'expression de la primitive $F$ de $f$ vérifiant $F(0)=\dfrac{\pi}{6}$
\end{itemize}
	\ifsolutions \else 
\vspace*{10cm}
\fi 
\eexo
\solution{
Les réponses sont:
\begin{itemize}
	\item $f(x)=x^3-2x^2-5x+6$
	\item $f'(x)=3x^2-4x-5$
	\item $F(x)=\dfrac{x^4}{4}-\dfrac{2x^3}{3}-\dfrac{5 x^2}{2}+6x+\dfrac{\pi}{6}$
\end{itemize}
}

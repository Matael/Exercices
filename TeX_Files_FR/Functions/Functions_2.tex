\bexo
Tracer sur la même figure
\begin{itemize}
	\item $x\mapsto \tr{arctan}(x)$
	\item $x\mapsto \tr{cotan}(x)$
\end{itemize}
\eexo
\solution{
Les graphes des fonctions sont:\\
\begin{center}
	\begin{tikzpicture}
	  \begin{axis}[line width=2pt,       legend columns=-1,   title=Fonctions à tracer,   xlabel={$x$},   ylabel={$y$}, legend entries={$\tr{atan}(x)$,$\tr{cot}(x)$},
	  legend to name=named,ymax=3,ymin=-3]
		\addplot[blue,samples=500] expression[domain=-pi:pi,samples=500]{(pi/180)*atan(x)};
		\addplot[red,samples=500] expression[domain=-pi:pi]{cot(180*x/pi)};			
	  \end{axis}
	\end{tikzpicture}\\
		  \ref{named}
\end{center}



}

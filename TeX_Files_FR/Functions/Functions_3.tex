\bexo
On considère la fonction
\begin{equation}
	f:x\mapsto (x-1)(x+2)(2x+3)
\end{equation}
\begin{itemize}
	\item Développez la fonction $f$
	\item Donner l'expression de la dérivée $f'$ de la fonction $f$.
	\item Donner l'expression de la primitive $F$ de $f$ vérifiant $F(0)=1$
\end{itemize}
\eexo
\solution{
Les réponses sont:
\begin{itemize}
	\item $f(x)=2x^3+5x^2-x-6$
	\item $f'(x)=6x^2+10x$
	\item $F(x)=\dfrac{1}{2}x^4+\dfrac{5}{3}x^2-\dfrac{x^2}{2}-6x+1$
\end{itemize}
}
